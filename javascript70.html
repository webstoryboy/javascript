continue문

break문과 continue문은 반복문을 제어하는 명령어입니다.

continue문

코드 실행 중에 continue을 만나면, 뒷 부분은 실행하지 않고 다음 반복 시퀀스로 넘어갑니다.

for(초깃값; 조건식; 증감식){
     continue;
     실행문
}

var 변수 = 초깃값;
while(조건식){
     continue;
     실행문;
     증감식;
}

break문와 continue문을 이용한 예제입니다.

html
<html lang="ko">

	<title>Javascript</title>
	<script>
		//while문을 이용해서 1~20까지 출력해주세요!
		// 숫자 1
		// 숫자 2
		// ...........
		// 숫자 20
		let num = 0;
		while (num < 20){
			num++;
			document.write("숫자 " + num, "<br>");

		//for문을 이용해서 1~20까지 출력해주세요!
		// 숫자 1
		// 숫자 2
		// ...........
		// 숫자 20
		for (var i=1; i<=20; i++){
			document.write("숫자 " + i, "<br>");
		}

		//break문과 continue문
		var count = 0;
		while(true){
			count++;
		if( count == 3){
			continue;
		}
		document.write("카운트" + count, "<br>");

		if ( count > 19 ){
			break;
			}
		}
	</script>
