화살표 함수를 사용하는 기본 예제입니다.

결과
1. 함수가 실행되었습니다.

html
<!DOCTYPE html>
<html lang="ko">
<head>
    <meta charset="UTF-8">
    <title>클로저</title>
    <script>
        function func10(){
            var text = "1. 함수가 실행되었습니다.<br>";
            return text;
        }
        var result = func10();
        document.write(result);
        
        function func11(){
            var title = "2. 함수가 실행되었습니다.";
            return function(){
                document.write(title);
            };
        }
        var result2 = func11();
        result2();
    </script>
</head>
