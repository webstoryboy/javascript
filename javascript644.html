가변인자 함수

가변인자 함수는 매개 변수의 지정없이 사용하는 함수입니다. arguments는 가상 공간에 변수 값을 지정하여 배열 형태로 사용합니다.

        function func1() {
            var num = arguments.length;
            var sum = 0;
            for(var i=0; i<num; i++) {
                sum = sum + arguments[i];
            }
            return sum;
        }
        document.write(func1(1,2,3,4,5),"<br>");
        
        function func2() {
            var sum = 0;
            for(var i in arguments) {
                sum += arguments[i];
            }
            return sum;
        }

        document.write(func2(1,2,3,4,5));
    </script>
